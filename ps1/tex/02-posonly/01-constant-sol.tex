	\frac{p(y^{(i)}=1,x^{(i)}|t^{(i)}=1)p(t^{(i)}=1)}{p(x^{(i)})p(y^{(i)}=1|t^{(i)}=1)}
	=\frac{p(y^{(i)}=1|x^{(i)})}{p(y^{(i)}=1|t^{(i)}=1)}
\end{align*}
the last but one $=$ holds by the assumption.
